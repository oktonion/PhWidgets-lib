52  //! Condition variable class.
53  //! This is a signaling object for synchronizing the execution flow for
54  //! several threads. Example usage:
55  //! @code
56  //! // Shared data and associated mutex and condition variable objects
57  //! int count;
58  //! mutex m;
59  //! condition_variable cond;
60  //!
61  //! // Wait for the counter to reach a certain number
62  //! void wait_counter(int targetCount)
63  //! {
64  //! lock_guard<mutex> guard(m);
65  //! while(count < targetCount)
66  //! cond.wait(m);
67  //! }
68  //!
69  //! // Increment the counter, and notify waiting threads
