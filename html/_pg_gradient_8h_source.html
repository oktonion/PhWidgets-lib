63 typedef struct _Pg_gradientcontrol {
64 
65  int gradient_type; /* Pg_GRAD_NOGRADIENT, Pg_GRAD_VECTOR, Pg_GRAD_RADIAL etc. */
66 
67  PhPoint_t start_point; /* The start-point of the gradient vector / the center for radial gradients */
68  /* the start-point is recomputed if a rotation is specified. */
69 
70  int ncolors; /* number of isochrome points = length of the gradient vector in pixels */
71  PgColor_t *colors; /* The isochrome colors: colors[0]=start_color .... colors[ncolors-1]=end_color */
72  /* This is the "gradient palette". */
73 
74  int *lookup; /* Lookup table. Contains indecies into the colors[] array. */
75  int lookup_size; /* Number of entries in the lookup table */
76  int xlookup; /* 1- the lookup table is indexed by X distance (from the normal line at the start point) */
77  /* 0- the lookup table is indexed by Y distance */
78 
79  int top_color_index; /* Index of the color above the gradient band (0 or ncolors-1) */
80  int bottom_color_index; /* Index of the color below the gradient band (0 or ncolors-1) */
81 
82  double slope; /* Slope of the normal line */
83  double intercept; /* Intercept of the normal line at the start point */
84  double intercept1; /* Intercept of the normal line at the end point */
85 
86  double *intabove; /* Intercept of the top line in the gradient band (&intercept or &intercept1) */
87  double *intbelow; /* Intercept of the bottom line in the gradient band (&intercept or &intercept1) */
88 
89  double vslope; /* Slope of the gradient vector itself == -1/slope */
90  double vintercept; /* Intercept of the gradient vector */
91 
92 } PgGradientControl_t;
