37 /*--------------------------------------------------------
38  * Confirm callback function
39  *--------------------------------------------------------
40  * node - the current xml node
41  * tdata - Tree item data for this node. The confirm function should set the
42  * appropriate members. The 'display_name' member is the 'display_name'
43  * parameter of XML_PtTreePopulate(), it is ignored if XML_PF_TAGNAME is set

45  * initialized to -1.
46  * depth - the depth of this entry in the tree/menu (where 0 is the root-level).
47  * cbdata - arbitrary callback data which was passed to XML_PtTreePopulate()
48  *
49  * Returns: 0 to skip the node, or any of the "confirm callback return flags" (below).
50  */

62 /*--------------------------------------------------------
63  * Confirm callback return flags
64  *--------------------------------------------------------
65  */
66 #define XML_PF_ADD 0x0001 /* Add the node. */
67 #define XML_PF_CHILDREN 0x0002 /* Process the children of this node. */
68 #define XML_PF_ADD_IF_HAS_CHILDREN 0x0007 /* Add the node. Process its children. Remove node if it */
69  /* does not end up with any children or if it is excluded */
70  /* by the duplicate-handling mechanism. */
71 #define XML_PF_TAGNAME 0x0008 /* Display the tagname (ignore the display_name buffer ) */
72 #define XML_PF_EXPAND 0x0010 /* Expand this node once all its children have been created. */
73 #define XML_PF_SORT 0x0020 /* Sort the children of this node alphabetically in the PtTree. */
74  /* The sort is case sensitive. */
75 #define XML_PF_SORTI 0x0040 /* Case insensitive version of XML_PF_SORT. Takes precedence */
76  /* over XML_PF_SORT if both sort-flags are specified. */
77 #define XML_PF_DUPI 0x0080 /* When the children of this node are processed, use a case */
78  /* insensitive search for duplicates. */
79  /* The default is a case sensitive search. */

125 /*--------------------------------------------------------
126  * XML_PtTreePopulate flags
127  *--------------------------------------------------------
128  *
129  * XML_PF_SORT Sort the children of troot alphabetically.
130  * The sort is case sensitive.
131  * If confirm_fn is NULL, then the tree-nodes will be sorted at every level,
132  * otherwise, the confirm_fn decides the sorting policy at each level
133  * via the flag it returns.
134  * XML_PF_SORTI Case insensitive version of XML_PF_SORT.
135  * If both sort-flags are specified, then XML_PF_SORTI takes precedence.
136  * XML_PF_DUPI When the children of xroot are processed, use a case insensitive search
137  * for duplicates. The default is a case sensitive search.
138  * If confirm_fn is NULL, then this flag will be in effect at
139  * every level, otherwise, the confirm_fn decides the case sensitivity
140  * of the duplicate search at each level via the flag it returns.
141 */

147 int XML_PtTreePopulate(PtWidget_t *tree, PtTreeItem_t *troot, xml_node_t *xroot,
148  XML_confirm_t confirm_fn, XML_duplicate_t duplic_fn, XML_create_t create_fn,
149  treetraverse_t destroy_fn, int flags, char *display_name, int starting_depth, void *cbdata);
150 
151  /* 'tree' is the PtTree widget to update.
152  * 'troot' is the item in the tree widget to update (can be NULL).
153  * 'xroot' is the root xml node.
154  * 'confirm_fn' is a callback to confirm whether to add a particular node
155  * (see XML_confirm_t function, above). If it's NULL, then all nodes are added.
156  * 'duplic_fn' is a callback which is called when a duplicate tree item is
157  * detected (see XML_duplicate_t and "Duplicate callback return values" above).
158  * 'create_fn' is a callback which is called when a new tree node is being
159  * added, so that you can associate arbitrary user-data to the tree node.
160  * (see XML_create_t, above).
161  * The normal user data to be put into this node is a pointer to the XML node,
162  * which is done by having the 'create_fn' return the 'node' parameter.
163  * 'destroy_fn' is a callback which is called when the tree node is being
164  * destroyed, so that you can deallocate any memory associated with the
165  * user-data for that node.
166  * 'flags' are made up of the XML_PtTreePopulate flags. above.
167  * 'display_name' is a character buffer for the display_name that will be
168  * returned to confirm_fn in the 'display_name' member of xml_tree_data_t.
169  * 'starting_depth' should be 0 ('depth' parameter for the root of the tree).
170  * 'cbdata' is arbitrary callback data which will be passed to each of the
171  * callback functions.
172  * Returns: 0 on success, -1 on error.
173  */
174

178  /* Fills a PtList or PtComboBox with the elements of the given 'nodelist'.
179  * 'widget' is a pointer to the PtComboBox.
180  * 'nodelist' is the list which has been previously initialized, and filled with a number of nodes.
181  * 'selected_node' is a pointer to the node in the 'nodelist' which is to be selected.
182  * Set 'style' to one of the following XML_COMBO_STYLE_* constants:
183  * XML_COMBO_STYLE_TAGNAME uses each node's tag-name as the combo-box entry.
184  * XML_COMBO_STYLE_TEXTDATA uses each node's text data as the combo-box entry.
185  * XML_COMBO_STYLE_ATTRIBUTE uses each node's specified attribute as the combo-box entry.
186  * 'attribute' is a string specifying the attribute to use, if XML_COMBO_STYLE_ATTRIBUTE is used.
187  * Returns -1 if an error is detected. Otherwise, it returns the element of 'nodelist' which is
188  * originally selected in the ComboBox (0-based).
189  */

193 PtWidget_t *XML_PtMenuPopulate(const char *menu_title, PtWidget_t *parent, xml_node_t *node,
194  XML_confirm_t confirm_fn, XML_activate_t activate_fn, char *display_name,
195  int starting_depth, void *cbdata);

197  /* 'menu_title' is the title that will appear at the top of the new menu. If it is NULL,
198  * no title will be added to the menu.
199  * 'parent' is the parent widget for the menu.
200  * 'node' is the root xml node
201  * 'confirm_fn' is a callback to confirm whether to add a particular node (see XML_PtTreePopulate.h)
202  * 'activate_fn' is the function to be called when the menu item is selected (see XML_Activate_t).
203  * Set 'depth' to -1 to recurse the entire tree.
204  * 'display_name' is a buffer, which is sent as a parameter to the confirmation function to be filled.
205  * The buffer must be a global string in your application, since it will be referenced at any
206  * time the menu is referenced, not only when XML_PtMenuPopulate is called.
207  * 'starting_depth' should be 0 ('depth' parameter for the root of the menu).
208  * 'cbdata' is arbitrary callback data to be sent to the callbacks.
209  * The function returns a pointer to the Menu widget, or NULL on error.
210  * You must set the position of the widget manually, after this function returns.
211  */

241  *
242  * void menucallback(PtWidget_t *widget, xml_node_t *node, void *data)

251  * if(node==NULL || data==NULL) return(0);
252  * strncpy(data->display_name, XML_NodeNameGet(node), 254);
253  * data->display_name[255]=0;
254  * return(ret);
255  * }

260  * PhPoint_t pos, window;
261  * short x, y;
262  * xml_node_t *node;
263  * char name_buffer[256];
264  *
